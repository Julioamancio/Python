Conexão com Bancos de Dados (SQLite, MySQL)

Para salvar dados permanentemente ou compartilhar entre programas, usamos bancos de dados. O sqlite3 é nativo do Python, leve e não precisa de servidor. O sqlalchemy é uma biblioteca que facilita trabalhar com banco de dados de forma mais profissional e compatível com vários tipos (MySQL, PostgreSQL, etc).

Exemplo 1 (SQLite básico): 
# Cria banco SQLite e salva uma pessoa
import sqlite3
conn = sqlite3.connect("meubanco.db")  # Cria/abre arquivo de banco
c = conn.cursor()                      # Cria cursor para comandos SQL
c.execute("CREATE TABLE IF NOT EXISTS pessoas (nome TEXT)")  # Cria tabela se não existir
c.execute("INSERT INTO pessoas VALUES ('Ana')")              # Insere registro
conn.commit()                        # Salva alterações
conn.close()                         # Fecha conexão
Exemplo 2 (Leitura com SQLite): 
# Lê registros do banco criado
import sqlite3
conn = sqlite3.connect("meubanco.db")
c = conn.cursor()
for linha in c.execute("SELECT * FROM pessoas"):  # Busca todos os registros
    print(linha)
conn.close()
Exemplo 3 (SQLAlchemy): 
# SQLAlchemy facilita trabalhar com diferentes bancos (ex: SQLite, MySQL)
from sqlalchemy import create_engine, text
engine = create_engine("sqlite:///meubanco.db")  # Conexão compatível com SQLAlchemy
with engine.connect() as conn:
    conn.execute(text("CREATE TABLE IF NOT EXISTS teste (x INT)"))  # Executa comando SQL
Interfaces Gráficas (tkinter, PyQt)

Interfaces gráficas permitem criar programas com janelas, botões, caixas de texto e outros componentes visuais. tkinter é nativo do Python, enquanto PyQt é um framework mais avançado e profissional.

Exemplo 1 (Tkinter): 
# Janela com mensagem
import tkinter as tk
root = tk.Tk()
tk.Label(root, text="Olá!").pack()
root.mainloop()
Exemplo 2 (Tkinter botão): 
# Botão que imprime mensagem ao clicar
import tkinter as tk
def clique():
    print("Clicou!")
janela = tk.Tk()
tk.Button(janela, text="Clique", command=clique).pack()
janela.mainloop()
Exemplo 3 (PyQt5): 
# Janela PyQt5 (precisa instalar PyQt5)
from PyQt5.QtWidgets import QApplication, QLabel
app = QApplication([])
label = QLabel("Olá, PyQt!")
label.show()
app.exec()
Web Scraping (requests, BeautifulSoup)

Web scraping é a técnica de extrair informações de páginas da internet de forma automática. requests baixa o conteúdo, BeautifulSoup analisa o HTML e busca dados específicos.

Exemplo 1 (requests): 
# Baixa e mostra parte do HTML da página
import requests
resp = requests.get("https://www.example.com")
print(resp.text[:100])
Exemplo 2 (BeautifulSoup): 
# Mostra o título da página python.org
from bs4 import BeautifulSoup
import requests
html = requests.get("https://www.python.org").text
soup = BeautifulSoup(html, "html.parser")
print(soup.title.text)
Exemplo 3 (links com BeautifulSoup): 
# Lista todos os links da página
from bs4 import BeautifulSoup
import requests
html = requests.get("https://www.python.org").text
soup = BeautifulSoup(html, "html.parser")
for a in soup.find_all("a"):
    print(a.get("href"))
APIs REST (criação e consumo)

APIs REST permitem que programas conversem pela internet, usando HTTP. Você pode consumir APIs prontas (buscar dados) ou criar as suas (expor funcionalidades).

Exemplo 1 (consumindo API): 
# Usa requests para buscar dados de uma API pública
import requests
r = requests.get("https://jsonplaceholder.typicode.com/todos/1")
print(r.json())
Exemplo 2 (Flask API): 
# Cria uma API simples com Flask
from flask import Flask, jsonify
app = Flask(__name__)
@app.route("/api")
def api():
    return jsonify({"msg": "Hello"})
# app.run()
Exemplo 3 (requisição POST): 
# Envia dados para API simulada usando POST
import requests
r = requests.post("https://httpbin.org/post", data={"nome": "Julio"})
print(r.json())

Threads e Processos (threading, multiprocessing)

Threads e processos permitem executar tarefas em paralelo, melhorando a performance em tarefas pesadas ou demoradas. threading é para tarefas simultâneas leves, multiprocessing para processos realmente separados.

Exemplo 1 (thread): 
# Executa função em thread separada
import threading
def tarefa():
    print("Executando em thread")
t = threading.Thread(target=tarefa)
t.start()
t.join()
Exemplo 2 (múltiplas threads): 
# 3 threads simultâneas, imprimindo valores
import threading
for i in range(3):
    threading.Thread(target=lambda: print(i)).start()
Exemplo 3 (processo): 
# Processo separado do principal
from multiprocessing import Process
def tarefa():
    print("Processo novo!")
p = Process(target=tarefa)
p.start()
p.join()
Programação Assíncrona (async, await)

Programação assíncrona permite rodar várias tarefas que dependem de respostas externas (como internet) sem travar o programa. async define funções assíncronas, await espera uma resposta.

Exemplo 1: 
# Função assíncrona simples
import asyncio
async def main():
    print("Oi")
asyncio.run(main())
Exemplo 2: 
# Espera 1 segundo de forma não bloqueante
import asyncio
async def espera():
    await asyncio.sleep(1)
    print("Depois de 1 segundo")
asyncio.run(espera())
Exemplo 3: 
# Executa duas tarefas "ao mesmo tempo"
import asyncio
async def tarefa(i):
    await asyncio.sleep(i)
    print(f"Tarefa {i} pronta")
async def main():
    await asyncio.gather(tarefa(1), tarefa(2))
asyncio.run(main())
Ambientes Virtuais (venv, pip)

Ambientes virtuais isolam as bibliotecas de cada projeto, evitando conflitos e facilitando a manutenção. venv cria ambientes, pip instala bibliotecas apenas dentro do ambiente ativado.

Exemplo 1 (criar venv): 
# Cria ambiente virtual
python -m venv venv
# Ativar no Windows: venv\Scripts\activate
# Ativar no Linux/Mac: source venv/bin/activate
Exemplo 2 (instalar pacote): 
# Instala pacote para o projeto
pip install requests
pip list
Exemplo 3 (desativar ambiente): 
# Sai do ambiente virtual
deactivate
Controle de Dependências (requirements.txt, pip freeze)

Para garantir que todos os pacotes usados no projeto possam ser instalados em outro ambiente, use pip freeze para gerar um arquivo requirements.txt com as versões. Use esse arquivo para instalar tudo de uma vez.

Exemplo 1 (gerar requirements): 
# Gera lista de dependências instaladas
pip freeze > requirements.txt
Exemplo 2 (instalar todas): 
# Instala todas as dependências listadas
pip install -r requirements.txt
Exemplo 3 (conteúdo típico): 
# requirements.txt pode conter:
requests==2.31.0
Flask==3.0.2
Publicação de Pacotes (setuptools, PyPI)

Você pode empacotar seu projeto Python e publicar no PyPI (Python Package Index), permitindo que qualquer pessoa instale com pip. setuptools ajuda a empacotar, twine faz o upload.

Exemplo 1 (setup.py): 
# Script de configuração do seu pacote
from setuptools import setup
setup(name="meupacote", version="0.1", py_modules=["meupacote"])
Exemplo 2 (build e upload): 
# Gera os arquivos de distribuição e faz upload para o PyPI
python setup.py sdist bdist_wheel
twine upload dist/*
Exemplo 3 (instalar do PyPI): 
# Qualquer usuário pode instalar seu pacote
pip install nome-do-pacote
Desafio Final do Módulo Profissional

Desafio:

  • Crie um programa profissional que:
  • Crie um banco de dados SQLite para cadastro de clientes com nome e e-mail
  • Tenha uma interface gráfica simples (Tkinter) para cadastrar e listar clientes
  • Implemente função para exportar os dados para um arquivo clientes.csv
  • Use organização modular em funções ou classes
Solução sugerida (comentada): 
import sqlite3
import tkinter as tk
from tkinter import messagebox
import csv

# Cria banco e tabela se não existir
def criar_banco():
    conn = sqlite3.connect("clientes.db")
    c = conn.cursor()
    c.execute("CREATE TABLE IF NOT EXISTS clientes (nome TEXT, email TEXT)")
    conn.commit()
    conn.close()

# Salva um novo cliente
def salvar_cliente(nome, email):
    conn = sqlite3.connect("clientes.db")
    c = conn.cursor()
    c.execute("INSERT INTO clientes VALUES (?, ?)", (nome, email))
    conn.commit()
    conn.close()

# Retorna lista de todos os clientes
def listar_clientes():
    conn = sqlite3.connect("clientes.db")
    c = conn.cursor()
    c.execute("SELECT * FROM clientes")
    clientes = c.fetchall()
    conn.close()
    return clientes

# Exporta clientes para CSV
def exportar_csv():
    clientes = listar_clientes()
    with open("clientes.csv", "w", newline="") as f:
        writer = csv.writer(f)
        writer.writerow(["Nome", "Email"])
        writer.writerows(clientes)
    messagebox.showinfo("Exportação", "Exportado para clientes.csv!")

# Função chamada ao clicar em "Cadastrar"
def cadastrar():
    nome = entry_nome.get()
    email = entry_email.get()
    if nome and email:
        salvar_cliente(nome, email)
        entry_nome.delete(0, tk.END)
        entry_email.delete(0, tk.END)
        messagebox.showinfo("Cadastro", "Cliente salvo!")
        mostrar_clientes()
    else:
        messagebox.showwarning("Atenção", "Preencha todos os campos.")

# Atualiza a lista de clientes na interface
def mostrar_clientes():
    lista.delete(0, tk.END)
    for nome, email in listar_clientes():
        lista.insert(tk.END, f"{nome} - {email}")

criar_banco()
root = tk.Tk()
root.title("Cadastro de Clientes")

tk.Label(root, text="Nome:").pack()
entry_nome = tk.Entry(root)
entry_nome.pack()

tk.Label(root, text="E-mail:").pack()
entry_email = tk.Entry(root)
entry_email.pack()

tk.Button(root, text="Cadastrar", command=cadastrar).pack(pady=4)
tk.Button(root, text="Exportar CSV", command=exportar_csv).pack(pady=4)

lista = tk.Listbox(root, width=40)
lista.pack(pady=6)
mostrar_clientes()

root.mainloop()
